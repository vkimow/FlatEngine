cmake_minimum_required(VERSION 3.13 FATAL_ERROR)

project(Tools CXX)

set_flat_library_dirs(LIBRARY_NAME ${PROJECT_NAME})

set_flat_library_sources(  
    LIBRARY_NAME ${PROJECT_NAME}
    SOURCES ${FLAT_CURRENT_AUTO_SOURCES}
    )

set_flat_library_public_headers(    
    LIBRARY_NAME ${PROJECT_NAME}
    PUBLIC_HEADERS ${FLAT_CURRENT_AUTO_PUBLIC_HEADERS}
    )

add_flat_library(   
    LIBRARY_NAME ${PROJECT_NAME}
    SOURCES ${FLAT_CURRENT_SOURCES} ${FLAT_CURRENT_PUBLIC_HEADERS}
    INCLUDE_DIRS ${FLAT_CURRENT_INCLUDE_DIR} ${FLAT_CURRENT_SOURCE_DIR} ${SFML_INCLUDE_DIR}
    LINK_LIBS SFML
    )
