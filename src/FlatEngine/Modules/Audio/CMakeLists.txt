cmake_minimum_required(VERSION 3.13 FATAL_ERROR)

project(Audio CXX)

set_flat_library_dirs(
    LIBRARY_NAME ${PROJECT_NAME}
    INTERMEDIATE_FOLDERS ${FLAT_CURRENT_SOLUTION_FOLDER}
    )

set_flat_library_files(
    VARIABLE_NAME SOURCES
    LIBRARY_NAME ${PROJECT_NAME}
    FILES_DIR ${FLAT_CURRENT_SOURCE_DIR}
    FILES ${FLAT_CURRENT_AUTO_SOURCES}
    )

set_flat_library_files(
    VARIABLE_NAME PUBLIC_HEADERS
    LIBRARY_NAME ${PROJECT_NAME}
    FILES_DIR ${FLAT_CURRENT_INCLUDE_DIR}
    FILES ${FLAT_CURRENT_AUTO_PUBLIC_HEADERS}
    )

add_flat_library(   
    LIBRARY_NAME ${PROJECT_NAME}
    SOLUTION_FOLDER ${FLAT_CURRENT_SOLUTION_FOLDER}
    SOURCES ${FLAT_CURRENT_SOURCES} ${FLAT_CURRENT_PUBLIC_HEADERS}
    INCLUDE_DIRS ${FLAT_CURRENT_INCLUDE_DIR} ${FLAT_CURRENT_SOURCE_DIR} ${SFML_INCLUDE_DIR}
    LINK_LIBS SFML
    )
